# The Solution Of The Problem

1
2
3
4

  def solve(nums: Array[Int], k: Int): Array[Int] =
     nums.groupBy(identity)
     .view.mapValues(_.length)
     .toArray.sortBy(_._2)
     .map(_._1).reverse.take(k)
